Square Market Multiple Item Embedding with Async Script

If you prefer, you can include the script tag using appendChild so that it doesn't block page rendering:

    <script>
    (function(d, id, src) {
      var s = d.getElementById(id);
      if (!s) {
        s = d.createElement('script');
        s.id = id;
        s.src = src;
        d.head.appendChild(s);
      }
    })(document, "sq-embed-js", "https://cdn.sq-api.com/market/embed.js");
    </script>
